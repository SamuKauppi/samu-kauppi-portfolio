The Binomial distribution formula is given by: \( P(X = k) = \binom{n}{k} p^k (1 - p)^{n - k} \).

$$ P(X = k) = \binom{n}{k} p^k (1 - p)^{n - k} $$ 

            float AI_Player::binomial_coefficient(int n, int x) {
                if (x > n - x)
                    x = n - x;
            
                int coeff = 1; // Start as an integer
                for (int i = 0; i < x; ++i) {
                    coeff *= (n - i);
                    coeff /= (i + 1);
                }
            
                return static_cast(coeff); // Return as float
            }
